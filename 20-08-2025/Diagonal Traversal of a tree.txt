class Tree {
    public ArrayList<Integer> diagonal(Node root) {
         ArrayList<Integer> res=new ArrayList<>();
        if(root==null) return res;
        Queue<Node> q=new LinkedList<>();
        q.add(root);

        while (!q.isEmpty()) {
            Node node=q.poll();

            while(node!=null) {
                res.add(node.data);

                if(node.left!=null) {
                    q.add(node.left);
                }

                node=node.right;
            }
        }

        return res;
    }
}

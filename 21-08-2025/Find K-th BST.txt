class Solution {
     int c=0;
        int res=-1;
    public void inorder(TreeNode root,int k){
        if(root==null) return;
        inorder(root.left,k);
        c++;
        if(k==c){
            res=root.val;
            return ;
        }
        inorder(root.right,k);
    }
    public int kthSmallest(TreeNode root, int k) {
       inorder(root,k);
       return res;
       
    }
}
